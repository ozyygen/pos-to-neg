PREFIX wikibase: <http://wikiba.se/ontology#>
PREFIX wd: <http://www.wikidata.org/entity/>
PREFIX wdt: <http://www.wikidata.org/prop/direct/>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
SELECT ?sibling ?siblingLabel WHERE {
  # Find the subclass of the given constant
  <constant> wdt:P279 ?parentClass .
  
  # Find other entities with the same parent class
  ?sibling wdt:P279 ?parentClass .
  ?sibling rdfs:label ?siblingLabel. 
  FILTER (LANG(?siblingLabel) = "en") .
  
  # Exclude the original entity from the results
  FILTER(?sibling != <constant>)
  
 
}
